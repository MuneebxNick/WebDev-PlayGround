JavaScript Arrays

Arrays in JavaScript are used to store multiple values in a single variable. They are dynamic and can hold different data types.

Creating an Array

let fruits = ["Apple", "Banana", "Cherry"];
console.log(fruits); // Output: ["Apple", "Banana", "Cherry"]

Accessing Array Elements

console.log(fruits[0]); // Output: "Apple"
console.log(fruits[1]); // Output: "Banana"
console.log(fruits[2]); // Output: "Cherry"

Array Methods

push()

fruits.push("Orange");
console.log(fruits); // Output: ["Apple", "Banana", "Cherry", "Orange"]

pop()

fruits.pop();
console.log(fruits); // Output: ["Apple", "Banana", "Cherry"]

shift()

fruits.shift();
console.log(fruits); // Output: ["Banana", "Cherry"]

unshift()

fruits.unshift("Mango");
console.log(fruits); // Output: ["Mango", "Banana", "Cherry"]

length

console.log(fruits.length); // Output: 3

forEach()

fruits.forEach(function(item, index) {
    console.log(index + ": " + item);
});
// Output:
// 0: Mango
// 1: Banana
// 2: Cherry

map()

let upperFruits = fruits.map(function(item) {
    return item.toUpperCase();
});
console.log(upperFruits); // Output: ["MANGO", "BANANA", "CHERRY"]

filter()

let filteredFruits = fruits.filter(function(item) {
    return item.startsWith("B");
});
console.log(filteredFruits); // Output: ["Banana"]

find()

let foundFruit = fruits.find(function(item) {
    return item === "Cherry";
});
console.log(foundFruit); // Output: "Cherry"

sort()

fruits.sort();
console.log(fruits); // Output: ["Banana", "Cherry", "Mango"]

reverse()

fruits.reverse();
console.log(fruits); // Output: ["Mango", "Cherry", "Banana"]

Multidimensional Arrays

let matrix = [
    [1, 2, 3],
    [4, 5, 6],
    [7, 8, 9]
];
console.log(matrix[0][0]); // Output: 1
console.log(matrix[1][2]); // Output: 6
console.log(matrix[2][1]); // Output: 8

Conclusion

Arrays are a fundamental part of JavaScript and provide a powerful way to store and manipulate collections of data. With various built-in methods, you can easily perform operations like adding, removing, and iterating over elements.
